            #!/bin/bash

            ls -t "$@" | tail -n +1  > sorteradlista
            destination=/home/ola/website/brev/allabrevhtml
            (( x = 1 ))
            (( y = $(ls *docx | wc -l) ))
            while IFS= read -r line
            do
              n=$(printf %08d $x)
              pandoc --quiet -s -f docx -t html "$line" -o "$n${line:0:-5}.html"
              echo -ne "$x/$y \r"
              ((x ++ ))
            done < sorteradlista

            rm $destination/*
            mv *.html "$destination"

            > $destination/lista
            for f in $destination/*.html
            do
              filename=$(basename "$f")
              echo "<p><a href=\"allabrevhtml/$filename\">${filename:8:-5}<<p>" >> $destination/lista
            done

            sed -i '/allabrevhtml/d' /home/ola/website/brev/index.html
            sed -i '/letters go here/r /home/ola/website/brev/allabrevhtml/lista' /home/ola/website/brev/index.html
